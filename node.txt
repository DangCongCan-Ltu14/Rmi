CREATE TABLE mail ( hash VARCHAR(70) NOT NULL , id1 VARCHAR(70) NOT NULL
 , id2 VARCHAR(70) NOT NULL , 
tieu de VARCHAR(70) NULL , mess VARCHAR(500) NOT NULL , ngay DATE NOT NULL , 
gio TIME NOT NULL , PRIMARY KEY (hash)) ENGINE = InnoDB;

CREATE TABLE thongtin ( ten_cv VARCHAR(70) NOT NULL , ngay DATE NOT NULL , gio TIME NOT NULL , 
ghichu VARCHAR(500) NOT NULL , user VARCHAR(70) NOT NULL ) ENGINE = InnoDB;

CREATE TABLE user ( user VARCHAR(70) NOT NULL , pass VARCHAR(255) NOT NULL , PRIMARY KEY (user)) ENGINE = InnoDB;

ALTER TABLE thongtin ADD FOREIGN KEY(user) REFERENCES user(user);
